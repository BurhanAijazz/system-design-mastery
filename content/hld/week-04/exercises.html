import random

def jittered_ttl(base_ttl: int, jitter_pct: float = 0.1) -> int:
    """Add random jitter to TTL to prevent mass expiration."""
    jitter = int(base_ttl * jitter_pct)
    return base_ttl + random.randint(-jitter, jitter)
