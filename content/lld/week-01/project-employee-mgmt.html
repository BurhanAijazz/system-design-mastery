Project: Employee Management System Refactoring

In this project, you will take a monolithic employee management system that violates multiple SOLID principles and refactor it step by step into a clean, extensible architecture. This project tests your ability to apply all five SOLID principles in a realistic scenario.

Project Overview

  • Goal: Refactor the legacy EmployeeManager class into a SOLID-compliant design.
  • Skills tested: SRP, OCP, LSP, ISP, DIP
  • Estimated time: 2-3 hours
  • Deliverable: A working Python package with tests

The Legacy Code

Below is the monolithic system you must refactor. Read it carefully and identify every SOLID violation before you start coding. 

import json
import sqlite3
from datetime import datetime, timedelta


class EmployeeManager:
    """Legacy monolith: manages all employee operations."""

    def __init__(self):
        self.db = sqlite3.connect("employees.db")
        self._init_db()

    def _init_db(self):
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS employees (
                id INTEGER PRIMARY KEY,
                name TEXT, role TEXT, salary REAL,
                department TEXT, hire_date TEXT
            )
        """)

    def add_employee(self, name, role, salary, department):
        hire_date = datetime.now().isoformat()
        self.db.execute(
            "INSERT INTO employees (name, role, salary, department, hire_date) VALUES (?, ?, ?, ?, ?)",
            (name, role, salary, department, hire_date),
        )
        self.db.commit()
        self._send_welcome_email(name)

    def calculate_pay(self, employee_id):
        """Calculates monthly pay based on role."""
        emp = self._get_employee(employee_id)

        if emp["role"] == "full_time":
            pay = emp["salary"] / 12
        elif emp["role"] == "part_time":
            pay = emp["salary"] / 12 * 0.5
        elif emp["role"] == "contractor":
            pay = emp["salary"] / 12 * 1.1  # contractor premium
        elif emp["role"] == "intern":
            pay = emp["salary"] / 12 * 0.3

        # Tax calculation (SRP violation: pay calc + tax in one place)
        if pay > 10000:
            tax = pay * 0.30
        elif pay > 5000:
            tax = pay * 0.20
        else:
            tax = pay * 0.10

        return {"gross": pay, "tax": tax, "net": pay - tax}

    def generate_report(self, department, format_type):
        """Generate department report."""
        cursor = self.db.execute(
            "SELECT * FROM employees WHERE department = ?", (department,)
        )
        employees = cursor.fetchall()

        if format_type == "json":
            return json.dumps([dict(zip(["id", "name", "role", "salary", "dept", "hired"], e)) for e in employees])
        elif format_type == "csv":
            header = "id,name,role,salary,department,hire_date\n"
            rows = "\n".join(",".join(str(x) for x in e) for e in employees)
            return header + rows
        elif format_type == "summary":
            total_salary = sum(e[3] for e in employees)
            return f"Department: {department}, Employees: {len(employees)}, Total Salary: {total_salary}"

    def promote_employee(self, employee_id, new_role, salary_increase):
        emp = self._get_employee(employee_id)
        new_salary = emp["salary"] + salary_increase
        self.db.execute(
            "UPDATE employees SET role = ?, salary = ? WHERE id = ?",
            (new_role, new_salary, employee_id),
        )
        self.db.commit()
        self._log_action(f"Promoted {emp['name']} to {new_role}")
        self._send_promotion_email(emp["name"], new_role)

    def _get_employee(self, employee_id):
        cursor = self.db.execute("SELECT * FROM employees WHERE id = ?", (employee_id,))
        row = cursor.fetchone()
        if not row:
            raise ValueError(f"Employee {employee_id} not found")
        return dict(zip(["id", "name", "role", "salary", "department", "hire_date"], row))

    def _send_welcome_email(self, name):
        print(f"Sending welcome email to {name}")

    def _send_promotion_email(self, name, new_role):
        print(f"Sending promotion email: {name} is now {new_role}")

    def _log_action(self, message):
        with open("employee_actions.log", "a") as f:
            f.write(f"[{datetime.now()}] {message}\n")

Step 1: Identify Violations

Before coding, document every violation you find:

  • SRP violations: List each distinct responsibility in the class.
  • OCP violations: Find every if/elif chain that would grow with new requirements.
  • LSP concerns: Consider what would happen if you tried to create employee subtypes.
  • ISP concerns: Would a read-only reporting module need all these methods?
  • DIP violations: Find every hardcoded dependency (database, file, email).

Step 2: Design the Interfaces

Create abstractions for each dependency. Here is a suggested starting point: 

from abc import ABC, abstractmethod
from dataclasses import dataclass


@dataclass
class Employee:
    id: int | None
    name: str
    role: str
    salary: float
    department: str
    hire_date: str


class EmployeeRepository(ABC):
    @abstractmethod
    def save(self, employee: Employee) -> int:
        pass

    @abstractmethod
    def find_by_id(self, employee_id: int) -> Employee | None:
        pass

    @abstractmethod
    def find_by_department(self, department: str) -> list[Employee]:
        pass

    @abstractmethod
    def update(self, employee: Employee) -> None:
        pass


class PayCalculator(ABC):
    @abstractmethod
    def calculate_gross(self, employee: Employee) -> float:
        pass


class TaxCalculator(ABC):
    @abstractmethod
    def calculate_tax(self, gross_pay: float) -> float:
        pass


class NotificationService(ABC):
    @abstractmethod
    def send(self, recipient: str, subject: str, body: str) -> None:
        pass


class AuditLogger(ABC):
    @abstractmethod
    def log(self, action: str, details: str) -> None:
        pass


class ReportFormatter(ABC):
    @abstractmethod
    def format(self, employees: list[Employee]) -> str:
        pass

Step 3: Implement Pay Strategies (OCP)

class FullTimePayCalculator(PayCalculator):
    def calculate_gross(self, employee: Employee) -> float:
        return employee.salary / 12


class PartTimePayCalculator(PayCalculator):
    def calculate_gross(self, employee: Employee) -> float:
        return employee.salary / 12 * 0.5


class ContractorPayCalculator(PayCalculator):
    def calculate_gross(self, employee: Employee) -> float:
        return employee.salary / 12 * 1.1


class InternPayCalculator(PayCalculator):
    def calculate_gross(self, employee: Employee) -> float:
        return employee.salary / 12 * 0.3


class PayCalculatorRegistry:
    def __init__(self):
        self._calculators: dict[str, PayCalculator] = {}

    def register(self, role: str, calculator: PayCalculator):
        self._calculators[role] = calculator

    def get(self, role: str) -> PayCalculator:
        if role not in self._calculators:
            raise ValueError(f"No pay calculator for role: {role}")
        return self._calculators[role]

Step 4: Build the Refactored Services

class PayrollService:
    """Single responsibility: calculate pay for employees."""

    def __init__(self, calculator_registry: PayCalculatorRegistry, tax_calc: TaxCalculator):
        self._calculators = calculator_registry
        self._tax = tax_calc

    def calculate_pay(self, employee: Employee) -> dict:
        calculator = self._calculators.get(employee.role)
        gross = calculator.calculate_gross(employee)
        tax = self._tax.calculate_tax(gross)
        return {"gross": round(gross, 2), "tax": round(tax, 2), "net": round(gross - tax, 2)}


class EmployeeOnboardingService:
    """Single responsibility: onboard new employees."""

    def __init__(self, repo: EmployeeRepository, notifications: NotificationService, logger: AuditLogger):
        self._repo = repo
        self._notifications = notifications
        self._logger = logger

    def onboard(self, name: str, role: str, salary: float, department: str) -> Employee:
        from datetime import datetime
        employee = Employee(id=None, name=name, role=role, salary=salary, department=department, hire_date=datetime.now().isoformat())
        employee.id = self._repo.save(employee)
        self._notifications.send(name, "Welcome!", f"Welcome to {department}!")
        self._logger.log("onboard", f"New employee: {name}")
        return employee


class PromotionService:
    """Single responsibility: handle promotions."""

    def __init__(self, repo: EmployeeRepository, notifications: NotificationService, logger: AuditLogger):
        self._repo = repo
        self._notifications = notifications
        self._logger = logger

    def promote(self, employee_id: int, new_role: str, salary_increase: float) -> Employee:
        employee = self._repo.find_by_id(employee_id)
        if not employee:
            raise ValueError(f"Employee {employee_id} not found")
        employee.role = new_role
        employee.salary += salary_increase
        self._repo.update(employee)
        self._notifications.send(employee.name, "Promotion", f"You are now a {new_role}!")
        self._logger.log("promote", f"{employee.name} -> {new_role}")
        return employee

Requirements Checklist

  • No class should have more than one reason to change (SRP).
  • Adding a new employee role should not require modifying existing pay calculators (OCP).
  • All pay calculators must be substitutable through the PayCalculator interface (LSP).
  • Read-only report consumers should not depend on write operations (ISP).
  • No service should directly instantiate database connections or file handles (DIP).
  • Write at least 10 unit tests using fake/mock implementations.
  • All tests should pass without any real database, file system, or network.
