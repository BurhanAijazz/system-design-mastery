Null Object Pattern

The Null Object pattern provides a no-op (do-nothing) implementation of an interface instead of using None. This eliminates repetitive null checks throughout code.

The Problem: None Checks Everywhere

class OrderServiceNaive:
    """Every method must check if logger exists."""

    def __init__(self, logger=None):
        self._logger = logger

    def create_order(self, item: str) -> str:
        if self._logger is not None:  # Check 1
            self._logger.log(f"Creating order for {item}")

        order_id = f"ORD-{hash(item) % 1000:03d}"

        if self._logger is not None:  # Check 2
            self._logger.log(f"Order {order_id} created")

        return order_id

    def cancel_order(self, order_id: str) -> None:
        if self._logger is not None:  # Check 3
            self._logger.log(f"Cancelling {order_id}")

        # cancel logic...

        if self._logger is not None:  # Check 4
            self._logger.log(f"Order {order_id} cancelled")

Four null checks for just two methods. As the codebase grows, this explodes.

Solution: Null Object Pattern

Create a NullLogger that implements the same interface but does nothing:

from abc import ABC, abstractmethod

class Logger(ABC):
    """Logger interface."""

    @abstractmethod
    def log(self, message: str) -> None:
        pass

    @abstractmethod
    def error(self, message: str) -> None:
        pass


class ConsoleLogger(Logger):
    """Real logger that prints to console."""

    def log(self, message: str) -> None:
        print(f"[LOG] {message}")

    def error(self, message: str) -> None:
        print(f"[ERROR] {message}")


class NullLogger(Logger):
    """Null object -- does absolutely nothing.
    Same interface, zero behavior."""

    def log(self, message: str) -> None:
        pass  # intentionally empty

    def error(self, message: str) -> None:
        pass  # intentionally empty

Clean Service Code

class OrderService:
    """No None checks needed -- logger is always a valid object."""

    def __init__(self, logger: Logger = None):
        self._logger = logger or NullLogger()  # default to NullLogger

    def create_order(self, item: str) -> str:
        self._logger.log(f"Creating order for {item}")
        order_id = f"ORD-{hash(item) % 1000:03d}"
        self._logger.log(f"Order {order_id} created")
        return order_id

    def cancel_order(self, order_id: str) -> None:
        self._logger.log(f"Cancelling {order_id}")
        # cancel logic...
        self._logger.log(f"Order {order_id} cancelled")


# With real logging
svc = OrderService(ConsoleLogger())
svc.create_order("Widget")  # logs to console

# Without logging -- no None checks, no errors
svc = OrderService()  # uses NullLogger silently
svc.create_order("Gadget")  # NullLogger does nothing

Common Use Cases

  • Optional loggers -- NullLogger when logging is disabled
  • Default configurations -- NullConfig with sensible defaults
  • Missing entities -- NullUser for guest/anonymous users
  • Optional dependencies -- NullCache when caching is not configured
  • Placeholder objects -- NullRenderer during testing

Null Object vs None

  • None requires every caller to check before using
  • Null Object is a valid object that simply does nothing
  • Callers never need to check -- polymorphism handles it

When NOT to Use

Do not use Null Object when the absence of an object is semantically meaningful and callers should know about it. For example, a find_user() function should return None if the user truly does not exist and callers need to handle that case differently.
